class ImageLab:
    def __init__(self):
        self.pipeline_data = []

    def record(self, name, img):
        self.pipeline_data.append((name, img))

    def run_pipeline(self, img, funcs):

        current = img
        for name, func, params in funcs:
            current = func(current, **params)
            self.record(name, current)
